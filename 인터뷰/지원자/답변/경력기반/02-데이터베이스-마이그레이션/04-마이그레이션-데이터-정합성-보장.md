# 마이그레이션 중 데이터 정합성은 어떻게 보장했나요?

## 답변

금융 선불결제 시스템의 데이터베이스 마이그레이션에서 가장 중요한 것은 **단 한 건의 거래 데이터도 유실되거나 손상되지 않는 것**이었습니다. 특히 무중단 마이그레이션을 진행하면서 실시간으로 유입되는 수백만 건의 거래 데이터 정합성을 보장하는 것이 핵심 과제였습니다.

이를 위해 **3단계 검증 프로세스**(마이그레이션 전/중/후), **이중 쓰기(Dual Write) 전략**, **체크섬 기반 데이터 검증**, **단계별 롤백 계획**을 수립하고 실행했습니다.

특히 PostgreSQL에서 Oracle로의 마이그레이션 시에는 약 **3주간의 이중 쓰기 기간**을 운영하며 실시간으로 데이터 정합성을 검증했고, 전체 데이터(약 15억 건)의 **MD5 체크섬 검증**을 통해 100% 일치를 확인한 후 최종 전환했습니다.

## 핵심 키워드

- 데이터 정합성

- 이중화 전략
- 롤백 계획

### 2. 마이그레이션 중 검증

**1) 이중 쓰기(Dual Write) 전략**

```java
// 애플리케이션 레벨에서 양쪽 DB에 동시 쓰기
@Service
public class PaymentTransactionService {

    @Autowired
    private PostgresRepository postgresRepo;  // 기존 DB (Primary)

    @Autowired
    private OracleRepository oracleRepo;      // 신규 DB (Shadow)

    @Value("${migration.dual-write.enabled}")
    private boolean dualWriteEnabled;

    @Transactional
    public PaymentTransaction createTransaction(PaymentRequest request) {
        // 1. PostgreSQL에 저장 (Primary Write)
        PaymentTransaction result = postgresRepo.save(request);

        // 2. Oracle에도 동시 저장 (Shadow Write, 비동기)
        if (dualWriteEnabled) {
            CompletableFuture.runAsync(() -> {
                try {
                    oracleRepo.save(request);
                } catch (Exception e) {
                    // Oracle 쓰기 실패 시 로그만 남기고 계속 진행
                    // Primary(PostgreSQL)는 영향받지 않음
                    log.error("Shadow write to Oracle failed: {}", e.getMessage());
                    // 실패 건수를 별도 테이블에 기록하여 나중에 재처리
                    migrationFailureRepo.save(new MigrationFailure(
                        result.getTransactionId(),
                        "ORACLE_WRITE_FAILED",
                        e.getMessage()
                    ));
                }
            });
        }

        return result;
    }
}
```

```sql
FROM payment_transactions
```

```python
#!/usr/bin/env python3
import psycopg2
import cx_Oracle
import pandas as pd
from datetime import datetime, timedelta

def validate_data_consistency():
    # PostgreSQL 연결
    pg_conn = psycopg2.connect("host=pg-host dbname=payment_db user=postgres")
    pg_cursor = pg_conn.cursor()

    # Oracle 연결
    ora_conn = cx_Oracle.connect("user/password@oracle-host:1521/ORCL")
    ora_cursor = ora_conn.cursor()

    # 최근 2시간 데이터 비교
    validation_results = []

    for hour_offset in range(0, 48):  # 48시간치 검증
        target_hour = datetime.now() - timedelta(hours=hour_offset)

        # PostgreSQL 통계
        pg_cursor.execute("""
            SELECT COUNT(*), SUM(amount)
            FROM payment_transactions
            WHERE created_at >= %s AND created_at < %s
        """, (target_hour, target_hour + timedelta(hours=1)))
        pg_count, pg_sum = pg_cursor.fetchone()

        # Oracle 통계
        ora_cursor.execute("""
            SELECT COUNT(*), SUM(amount)
            FROM payment_transactions
            WHERE created_at >= :start_time AND created_at < :end_time
        """, start_time=target_hour, end_time=target_hour + timedelta(hours=1))
        ora_count, ora_sum = ora_cursor.fetchone()

        # 일치 여부 확인
        match = (pg_count == ora_count and
                 abs((pg_sum or 0) - (ora_sum or 0)) < 0.01)  # 부동소수점 오차 허용

        validation_results.append({
            'hour': target_hour,
            'pg_count': pg_count,
            'ora_count': ora_count,
            'count_match': pg_count == ora_count,
            'pg_sum': pg_sum,
            'ora_sum': ora_sum,
            'sum_match': abs((pg_sum or 0) - (ora_sum or 0)) < 0.01,
            'overall_match': match
        })

        if not match:
            print(f"[ALERT] Data mismatch at {target_hour}")
            print(f"  PostgreSQL: count={pg_count}, sum={pg_sum}")
            print(f"  Oracle:     count={ora_count}, sum={ora_sum}")
            # Slack/이메일 알림 발송

    # 결과를 검증 테이블에 저장
    df = pd.DataFrame(validation_results)
    # ... 저장 로직

    return all(r['overall_match'] for r in validation_results)

# 매시간 크론으로 실행
if __name__ == "__main__":
    if validate_data_consistency():
        print("✓ All data validated successfully")
    else:
        print("✗ Data validation failed - check alerts")
```

```sql
-- 전체 테이블 체크섬 생성 (파티션별로 수행)

-- PostgreSQL:
CREATE OR REPLACE FUNCTION calculate_partition_checksum(p_date DATE)
RETURNS TEXT AS $$
DECLARE
    v_checksum TEXT;
BEGIN
    SELECT MD5(
        STRING_AGG(
            transaction_id || '|' ||
            user_id || '|' ||
            amount || '|' ||
            status || '|' ||
            TO_CHAR(created_at, 'YYYY-MM-DD HH24:MI:SS'),
            '||'
            ORDER BY transaction_id
        )
    )
    INTO v_checksum
    FROM payment_transactions
    WHERE DATE(created_at) = p_date;

    RETURN v_checksum;
END;
$$ LANGUAGE plpgsql;

-- Oracle:
CREATE OR REPLACE FUNCTION calculate_partition_checksum(p_date DATE)
RETURN VARCHAR2 AS
    v_checksum VARCHAR2(32);
BEGIN
    SELECT LOWER(
        DBMS_CRYPTO.HASH(
            UTL_RAW.CAST_TO_RAW(
                LISTAGG(
                    transaction_id || '|' ||
                    user_id || '|' ||
                    amount || '|' ||
                    status || '|' ||
                    TO_CHAR(created_at, 'YYYY-MM-DD HH24:MI:SS'),
                    '||'
                ) WITHIN GROUP (ORDER BY transaction_id)
            ),
            2  -- MD5
        )
    )
    INTO v_checksum
    FROM payment_transactions
    WHERE TRUNC(created_at) = p_date;

    RETURN v_checksum;
END;
```

```bash
#!/bin/bash
# 파티션별로 데이터 덤프 후 diff 비교

for date in $(seq 0 365); do
    target_date=$(date -d "$date days ago" +%Y-%m-%d)

    # PostgreSQL 덤프
    psql -U postgres -d payment_db -c "
        COPY (
            SELECT * FROM payment_transactions
            WHERE DATE(created_at) = '$target_date'
            ORDER BY transaction_id
        ) TO '/tmp/pg_${target_date}.csv' CSV
    "

    # Oracle 덤프
    sqlplus user/pass@ORCL <<EOF
        SET COLSEP ','
        SET PAGESIZE 0
        SET TRIMSPOOL ON
        SET HEADSEP OFF
        SET LINESIZE 32767
        SPOOL /tmp/ora_${target_date}.csv
        SELECT * FROM payment_transactions
        WHERE TRUNC(created_at) = DATE '$target_date'
        ORDER BY transaction_id;
        SPOOL OFF
        EXIT
EOF

    # diff 비교
    diff /tmp/pg_${target_date}.csv /tmp/ora_${target_date}.csv
    if [ $? -eq 0 ]; then
        echo "✓ $target_date: Match"
    else
        echo "✗ $target_date: MISMATCH!"
        # 알림 발송
    fi
done
```

## 롤백 계획

### 1. 단계별 롤백 전략

```
Phase 1: 이중 쓰기 (3주)
- Primary: PostgreSQL (읽기/쓰기)
- Shadow: Oracle (쓰기만)
- 롤백: Oracle 쓰기 중단 (애플리케이션 설정 변경만)
- 영향도: 없음

Phase 2: 부분 전환 (1주)
- Primary: PostgreSQL (쓰기), Oracle (읽기 10%)
- Shadow: 없음
- 롤백: 읽기 트래픽을 PostgreSQL로 전환 (로드밸런서 설정)
- 영향도: 매우 낮음 (읽기만 영향)

Phase 3: 완전 전환
- Primary: Oracle (읽기/쓰기)
- Shadow: PostgreSQL (읽기 전용, 1주간 보관)
- 롤백: 쓰기를 PostgreSQL로 되돌림
- 영향도: 중간 (긴급 롤백 시 최대 10분 다운타임 예상)
- 완전 롤백 가능 기간: 전환 후 1주일
```

### 2. 롤백 실행 절차

```bash
#!/bin/bash
# 긴급 롤백 스크립트 (Oracle → PostgreSQL)

echo "=== EMERGENCY ROLLBACK INITIATED ==="

# 1. 읽기 트래픽 PostgreSQL로 전환 (즉시)
echo "Step 1: Switching read traffic to PostgreSQL..."
kubectl set env deployment/payment-api \
    DB_READ_ENDPOINT=postgresql-service.default.svc.cluster.local

# 2. 쓰기 트래픽 PostgreSQL로 전환
echo "Step 2: Switching write traffic to PostgreSQL..."
kubectl set env deployment/payment-api \
    DB_WRITE_ENDPOINT=postgresql-service.default.svc.cluster.local \
    DB_DUAL_WRITE_ENABLED=false

# 3. Oracle에서 마지막 트랜잭션 확인
echo "Step 3: Checking last transaction on Oracle..."
last_txn_id=$(sqlplus -s user/pass@ORCL <<EOF
    SET HEADING OFF
    SET FEEDBACK OFF
    SELECT MAX(transaction_id) FROM payment_transactions;
    EXIT
EOF
)

echo "Last transaction on Oracle: $last_txn_id"

# 4. PostgreSQL에 누락 데이터 복제 (Oracle → PostgreSQL)
echo "Step 4: Syncing missing data from Oracle to PostgreSQL..."
# Oracle에만 있고 PostgreSQL에 없는 데이터 복사
# (이중 쓰기 중 실패한 건)

# 5. 데이터 정합성 검증
echo "Step 5: Validating data consistency..."
python3 /scripts/validate_consistency.py

# 6. 애플리케이션 헬스체크
echo "Step 6: Checking application health..."
kubectl get pods -l app=payment-api
```

## 실제 발생한 이슈와 해결

### 이슈 1: 타임존 불일치

**문제:**
```sql
-- PostgreSQL: created_at = '2024-01-15 14:30:00+09' (KST, UTC+9)
-- Oracle: created_at = '2024-01-15 05:30:00' (UTC 저장됨)

-- 9시간 차이 발생!
```

**해결:**
```sql
-- Oracle에서 타임존 명시적 처리
ALTER SESSION SET TIME_ZONE = 'Asia/Seoul';

-- 애플리케이션 레벨에서 통일
spring.jpa.properties.hibernate.jdbc.time_zone=Asia/Seoul
```

### 이슈 2: Decimal 정밀도 손실

**문제:**
```sql
-- PostgreSQL: amount = 10000.12345678 (NUMERIC(15,8))
-- Oracle: amount = 10000.12345678 (NUMBER(15,8))

-- 부동소수점 연산 시 미세한 차이 발생
-- SUM(amount)가 0.03원 차이 발생 (15억 건 누적)
```

**해결:**
```sql
-- 금융 데이터는 항상 정수로 저장 (원 단위 또는 전 단위)
-- 10000.12 원 → 1000012 전 (100배)

ALTER TABLE payment_transactions
    ALTER COLUMN amount TYPE BIGINT;  -- 전 단위로 저장

-- 애플리케이션 레벨에서 변환
@Column(name = "amount")
private Long amountInJeon;  // 전 단위

public BigDecimal getAmount() {
    return new BigDecimal(amountInJeon).divide(BigDecimal.valueOf(100));
}
```

### 이슈 3: 시퀀스 불일치

**문제:**
```
PostgreSQL 마지막 ID: 1,523,456,789
Oracle 시퀀스 시작: 1

→ ID 충돌 발생!
```

**해결:**
```sql
-- Oracle 시퀀스를 PostgreSQL 마지막 값+1로 시작
CREATE SEQUENCE seq_payment_transaction
    START WITH 1523456790
    INCREMENT BY 1
    CACHE 1000;
```

### 이슈 4: 이중 쓰기 지연

**문제:**
```
초당 3000건 INSERT 시 Oracle 쓰기 지연 발생
PostgreSQL: 평균 15ms
Oracle: 평균 85ms

→ 비동기 큐가 쌓이면서 메모리 부족
```

**해결:**
```java
// 배치 쓰기로 전환
@Scheduled(fixedDelay = 1000)  // 1초마다
public void flushOracleQueue() {
    List<PaymentTransaction> batch = oracleQueue.poll(1000);  // 최대 1000건
    if (!batch.isEmpty()) {
        oracleRepo.saveAll(batch);  // JDBC batch insert
    }
}

// JDBC 배치 설정
spring.jpa.properties.hibernate.jdbc.batch_size=100
spring.jpa.properties.hibernate.order_inserts=true
```

### 이슈 5: 최종 전환 시 동기화 지연

**문제:**
```
최종 전환 직전 마지막 동기화 시간이 예상(30분)보다 오래 걸림 (2시간)
→ 유지보수 시간 초과 위험
```

**해결:**
```sql
-- 파티션별 병렬 동기화
-- 30개 파티션을 동시에 동기화 (파티션 프루닝 활용)

-- 병렬 작업 생성 (각 파티션마다)
parallel -j 30 psql -c "
    COPY (
        SELECT * FROM payment_transactions_y2024m01d{1}
    ) TO STDOUT
" | sqlplus user/pass@ORCL @load_partition_{1}.sql

-- 동기화 시간: 2시간 → 20분으로 단축
```
