## 백엔드 엔드포인트 방식

### 아키텍처

```
[브라우저] → [백엔드 API] → [Logstash/Elasticsearch]
     ↓
  로컬 버퍼
  (메모리)
```

### 동작 흐름

1. 브라우저에서 로그 이벤트 발생
2. 로컬 버퍼에 로그 저장
3. 일정 주기 또는 버퍼 크기 초과시 백엔드로 배치 전송
4. 백엔드에서 검증 및 변환
5. Logstash 또는 Elasticsearch로 전달

### React 구현

#### 1. 로그 서비스 유틸리티

```javascript
// src/utils/LogService.js

class LogService {
  constructor(backendUrl) {
    this.backendUrl = backendUrl;
    this.buffer = [];
    this.maxBufferSize = 10;
    this.flushInterval = 5000; // 5초

    this.startAutoFlush();
  }

  /**
   * 로그 기록
   * @param {string} level - 로그 레벨 (INFO, WARN, ERROR)
   * @param {string} message - 로그 메시지
   * @param {object} context - 추가 컨텍스트
   */
  log(level, message, context = {}) {
    const logEntry = {
      timestamp: new Date().toISOString(),
      level,
      message,
      context: {
        ...context,
        userAgent: navigator.userAgent,
        url: window.location.href,
        sessionId: this.getSessionId(),
        appVersion: process.env.REACT_APP_VERSION
      }
    };

    this.buffer.push(logEntry);

    // 버퍼가 가득 차면 즉시 전송
    if (this.buffer.length >= this.maxBufferSize) {
      this.flush();
    }
  }

  /**
   * 버퍼의 로그를 백엔드로 전송
   */
  async flush() {
    if (this.buffer.length === 0) return;

    const logs = [...this.buffer];
    this.buffer = [];

    try {
      const response = await fetch(`${this.backendUrl}/api/logs`, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${this.getAuthToken()}`
        },
        body: JSON.stringify({ logs })
      });

      if (!response.ok) {
        throw new Error(`HTTP ${response.status}`);
      }
    } catch (error) {
      console.error('Failed to send logs:', error);
      // 실패한 로그는 다시 버퍼에 추가 (최대 3번까지만)
      const retriableLogs = logs.filter(log =>
        !log._retryCount || log._retryCount < 3
      );
      retriableLogs.forEach(log => {
        log._retryCount = (log._retryCount || 0) + 1;
      });
      this.buffer.unshift(...retriableLogs);
    }
  }

  /**
   * 자동 플러시 시작
   */
  startAutoFlush() {
    setInterval(() => this.flush(), this.flushInterval);

    // 페이지 종료 시 남은 로그 전송
    window.addEventListener('beforeunload', () => {
      if (this.buffer.length > 0) {
        // sendBeacon은 비동기로 전송하며 페이지 종료를 막지 않음
        navigator.sendBeacon(
          `${this.backendUrl}/api/logs`,
          JSON.stringify({ logs: this.buffer })
        );
      }
    });
  }

  getSessionId() {
    let sessionId = sessionStorage.getItem('sessionId');
    if (!sessionId) {
      sessionId = this.generateSessionId();
      sessionStorage.setItem('sessionId', sessionId);
    }
    return sessionId;
  }

  generateSessionId() {
    return `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  }

  getAuthToken() {
    return localStorage.getItem('authToken') || '';
  }

  // 편의 메서드
  info(message, context) {
    this.log('INFO', message, context);
  }

  warn(message, context) {
    this.log('WARN', message, context);
  }

  error(message, context) {
    this.log('ERROR', message, context);
  }
}

export default new LogService(process.env.REACT_APP_LOG_API);
```

#### 2. React 컴포넌트에서 사용

```javascript
// src/App.js
import React, { useEffect } from 'react';
import logger from './utils/LogService';

const App = () => {
  useEffect(() => {
    // 전역 에러 리스너
    window.addEventListener('error', (event) => {
      logger.error(event.message, {
        stack: event.error?.stack,
        filename: event.filename,
        lineno: event.lineno,
        colno: event.colno,
        type: 'uncaughtError'
      });
    });

    // Unhandled Promise Rejection 리스너
    window.addEventListener('unhandledrejection', (event) => {
      logger.error('Unhandled Promise Rejection', {
        reason: event.reason,
        promise: event.promise,
        type: 'unhandledRejection'
      });
    });

    // 앱 시작 로그
    logger.info('Application started', {
      platform: navigator.platform,
      language: navigator.language
    });
  }, []);

  return (
    <div className="App">
      {/* Your app content */}
    </div>
  );
};

export default App;
```

#### 3. API 호출 로깅

```javascript
// src/utils/apiClient.js
import logger from './LogService';

export const apiClient = async (url, options = {}) => {
  const startTime = Date.now();

  try {
    const response = await fetch(url, options);
    const duration = Date.now() - startTime;

    // API 호출 로그
    logger.info('API call', {
      url,
      method: options.method || 'GET',
      status: response.status,
      duration
    });

    if (!response.ok) {
      logger.warn('API call failed', {
        url,
        status: response.status,
        statusText: response.statusText
      });
    }

    return response;
  } catch (error) {
    const duration = Date.now() - startTime;

    logger.error('API call error', {
      url,
      error: error.message,
      duration
    });

    throw error;
  }
};
```

### Angular 구현

```typescript
// src/app/services/logger.service.ts
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { environment } from '../../environments/environment';

interface LogEntry {
  timestamp: string;
  level: string;
  message: string;
  context?: any;
}

@Injectable({
  providedIn: 'root'
})
export class LoggerService {
  private buffer: LogEntry[] = [];
  private maxBufferSize = 10;
  private flushInterval = 5000;

  constructor(private http: HttpClient) {
    this.startAutoFlush();
  }

  log(level: string, message: string, context: any = {}) {
    const logEntry: LogEntry = {
      timestamp: new Date().toISOString(),
      level,
      message,
      context: {
        ...context,
        url: window.location.href,
        userAgent: navigator.userAgent
      }
    };

    this.buffer.push(logEntry);

    if (this.buffer.length >= this.maxBufferSize) {
      this.flush();
    }
  }

  private flush() {
    if (this.buffer.length === 0) return;

    const logs = [...this.buffer];
    this.buffer = [];

    this.http.post(`${environment.logApiUrl}/api/logs`, { logs })
      .subscribe({
        error: (err) => {
          console.error('Failed to send logs', err);
          // 재시도 로직
        }
      });
  }

  private startAutoFlush() {
    setInterval(() => this.flush(), this.flushInterval);
  }

  info(message: string, context?: any) {
    this.log('INFO', message, context);
  }

  warn(message: string, context?: any) {
    this.log('WARN', message, context);
  }

  error(message: string, context?: any) {
    this.log('ERROR', message, context);
  }
}
```

## 백엔드 API 구현

클라이언트에서 받은 로그를 Logstash로 전달하는 백엔드 예시입니다.

### Node.js + Express 구현

```javascript
// server.js
const express = require('express');
const winston = require('winston');
const LogstashTransport = require('winston-logstash-transport').LogstashTransport;
const rateLimit = require('express-rate-limit');
const helmet = require('helmet');

const app = express();

// 보안 미들웨어
app.use(helmet());
app.use(express.json({ limit: '1mb' }));

// Rate limiting
const limiter = rateLimit({
  windowMs: 15 * 60 * 1000, // 15분
  max: 100 // 최대 100개 요청
});

// Winston + Logstash 설정
const logger = winston.createLogger({
  transports: [
    new LogstashTransport({
      host: process.env.LOGSTASH_HOST || 'localhost',
      port: process.env.LOGSTASH_PORT || 5000
    }),
    new winston.transports.Console({
      format: winston.format.simple()
    })
  ]
});

// 로그 수신 엔드포인트
app.post('/api/logs', limiter, (req, res) => {
  const { logs } = req.body;

  // 입력 검증
  if (!Array.isArray(logs)) {
    return res.status(400).json({
      error: 'Invalid logs format. Expected array.'
    });
  }

  if (logs.length > 50) {
    return res.status(400).json({
      error: 'Too many logs. Maximum 50 per request.'
    });
  }

  // 각 로그 처리
  logs.forEach(log => {
    // 민감한 정보 필터링
    const sanitizedLog = sanitizeLog(log);

    // 추가 메타데이터
    const enrichedLog = {
      ...sanitizedLog,
      source: 'client-side',
      serverTimestamp: new Date().toISOString(),
      ip: req.ip,
      userAgent: req.get('user-agent')
    };

    // Logstash로 전송
    logger.log({
      level: sanitizedLog.level?.toLowerCase() || 'info',
      message: sanitizedLog.message,
      ...enrichedLog
    });
  });

  res.status(200).json({
    success: true,
    processed: logs.length
  });
});

/**
 * 민감한 정보 필터링
 */
function sanitizeLog(log) {
  const sanitized = { ...log };

  // 비밀번호, 토큰 등 제거
  const sensitiveFields = ['password', 'token', 'apiKey', 'secret'];

  function removeSensitive(obj) {
    if (typeof obj !== 'object' || obj === null) return;

    for (const key in obj) {
      if (sensitiveFields.some(field =>
        key.toLowerCase().includes(field.toLowerCase())
      )) {
        obj[key] = '[REDACTED]';
      } else if (typeof obj[key] === 'object') {
        removeSensitive(obj[key]);
      }
    }
  }

  removeSensitive(sanitized);
  return sanitized;
}

// 헬스체크
app.get('/health', (req, res) => {
  res.json({ status: 'ok' });
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`Log API server running on port ${PORT}`);
});
```

### Python + Flask 구현

```python
# app.py
from flask import Flask, request, jsonify
from flask_limiter import Limiter
from flask_limiter.util import get_remote_address
import logging
import logstash
from datetime import datetime

app = Flask(__name__)

# Rate limiting
limiter = Limiter(
    app=app,
    key_func=get_remote_address,
    default_limits=["100 per 15 minutes"]
)

# Logstash 핸들러 설정
logstash_handler = logstash.TCPLogstashHandler(
    host='localhost',
    port=5000,
    version=1
)

logger = logging.getLogger('client-logs')
logger.setLevel(logging.INFO)
logger.addHandler(logstash_handler)

@app.route('/api/logs', methods=['POST'])
@limiter.limit("100 per 15 minutes")
def receive_logs():
    data = request.get_json()

    if not isinstance(data.get('logs'), list):
        return jsonify({'error': 'Invalid logs format'}), 400

    logs = data['logs']

    if len(logs) > 50:
        return jsonify({'error': 'Too many logs'}), 400

    for log in logs:
        # 민감 정보 제거
        sanitized = sanitize_log(log)

        # 추가 메타데이터
        enriched = {
            **sanitized,
            'source': 'client-side',
            'server_timestamp': datetime.utcnow().isoformat(),
            'ip': request.remote_addr
        }

        # 로그 레벨에 따라 전송
        level = sanitized.get('level', 'INFO').upper()

        if level == 'ERROR':
            logger.error(sanitized.get('message'), extra=enriched)
        elif level == 'WARN':
            logger.warning(sanitized.get('message'), extra=enriched)
        else:
            logger.info(sanitized.get('message'), extra=enriched)

    return jsonify({
        'success': True,
        'processed': len(logs)
    }), 200

def sanitize_log(log):
    """민감한 정보 제거"""
    sensitive_fields = ['password', 'token', 'apikey', 'secret']

    def remove_sensitive(obj):
        if not isinstance(obj, dict):
            return

        for key in list(obj.keys()):
            if any(s in key.lower() for s in sensitive_fields):
                obj[key] = '[REDACTED]'
            elif isinstance(obj[key], dict):
                remove_sensitive(obj[key])

    sanitized = log.copy()
    remove_sensitive(sanitized)
    return sanitized

if __name__ == '__main__':
    app.run(port=3000)
```
